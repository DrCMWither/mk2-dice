/**
 * @typedef {Object} CommandSpec
 * @property {Function} handler - The function to execute when the command is triggered.
 * @property {string[]} [aliases] - Alternative names for the command.
 * @property {string} [parseMode] - The parse mode for Telegram messages (defaults to "HTML").
 * @property {Function|string|number} [targetChatId] - Optional logic to determine where the response is sent.
 */

/**
 * @typedef {Object} CallbackSpec
 * @property {string} name - The unique identifier for the callback type.
 * @property {RegExp|Function|string} match - The condition to match against callback data.
 * @property {Function} handler - The function to execute when the callback matches.
 */

/**
 * Manages the registration, aliasing, and dispatching of bot commands and callback queries.
 */
export class CommandRegistry {
    constructor() {
        /** @type {Map<string, CommandSpec>} */
        this.commands = new Map();
        /** @type {Map<string, string>} */
        this.aliases = new Map();
        /** @type {CallbackSpec[]} */
        this.callbacks = [];
    }

    /**
     * Normalizes a command or alias name to a trimmed, lowercase string.
     * @param {string} name - The raw name string.
     * @returns {string} The normalized name.
     */
    normalizeName(name) {
        return String(name || "").trim().toLowerCase();
    }

    /**
     * Registers a new command with its handler and optional configuration.
     * @param {string} name - The primary name of the command.
     * @param {CommandSpec|Function} spec - The command configuration or just the handler function.
     * @throws {TypeError} If the handler is missing or not a function.
     * @returns {this} The registry instance for chaining.
     */
    register(name, spec) {
        const primary = this.normalizeName(name);
        const command = typeof spec === "function" ? { handler: spec } : spec;

        if (!command || typeof command.handler !== "function") {
            throw new TypeError(`Command "${primary}" must provide a handler.`);
        }

        if (this.commands.has(primary)) {
            console.warn(`[WARN] Command "${primary}" re-registered, overriding old handler.`);
        }

        this.commands.set(primary, {
            name: primary,
            aliases: [],
            parseMode: "HTML",
            ...command,
        });

        for (const alias of command.aliases || []) {
            this.alias(alias, primary);
        }

        return this;
    }

    /**
     * Creates an alias for an existing primary command.
     * @param {string} aliasName - The new alias to register.
     * @param {string} primaryName - The existing primary command name.
     * @throws {Error} If the primary command does not exist.
     * @returns {this} The registry instance for chaining.
     */
    alias(aliasName, primaryName) {
        const alias = this.normalizeName(aliasName);
        const primary = this.normalizeName(primaryName);

        if (!this.commands.has(primary)) {
            throw new Error(`Cannot alias "${alias}" to unknown command "${primary}".`);
        }

        this.aliases.set(alias, primary);
        return this;
    }

    /**
     * Retrieves a command by its primary name or alias.
     * @param {string} name - The name or alias to look up.
     * @returns {CommandSpec|null} The command specification or null if not found.
     */
    get(name) {
        const key = this.normalizeName(name);
        const primary = this.aliases.get(key) || key;
        return this.commands.get(primary) || null;
    }

    /**
     * Parses a raw text message to see if it matches a registered command.
     * Supports bot-specific addressing (e.g., /roll@MyBot).
     * @param {string} text - The raw message text from Telegram.
     * @param {string} [botName] - The username of the bot to validate addressed commands.
     * @returns {Object|null} An object containing the command name, args, and spec, or null.
     */
    parseCommand(text, botName) {
        const input = String(text || "").slice(0, 1024);
        const match = input.match(/^\/([A-Za-z0-9_]+)(?:@([A-Za-z0-9_]+))?(?:\s+([\s\S]+))?$/);

        if (!match) return null;

        const [, rawName, addressedBot, args = ""] = match;

        if (
            botName &&
            addressedBot &&
            addressedBot.toLowerCase() !== String(botName).toLowerCase()
        ) {
            return null;
        }

        const name = this.normalizeName(rawName);
        const command = this.get(name);

        if (!command) return null;

        return {
            name,
            args,
            command,
        };
    }

    /**
     * Registers a callback query handler.
     * @param {string} name - Internal identifier for the callback.
     * @param {CallbackSpec} spec - The callback configuration.
     * @throws {TypeError} If the handler is missing or not a function.
     * @returns {this} The registry instance for chaining.
     */
    registerCallback(name, spec) {
        if (!spec || typeof spec.handler !== "function") {
            throw new TypeError(`Callback "${name}" must provide a handler.`);
        }

        this.callbacks.push({
            name,
            ...spec,
        });

        return this;
    }

    /**
     * Attempts to find a matching callback handler for given data.
     * @param {string} data - The callback data string.
     * @returns {Object|null} The matching callback and the regex match results (if applicable).
     */
    matchCallback(data) {
        for (const callback of this.callbacks) {
            if (callback.match instanceof RegExp) {
                const match = String(data || "").match(callback.match);
                if (match) return { callback, match };
            }

            if (typeof callback.match === "function" && callback.match(data)) {
                return { callback, match: null };
            }

            if (typeof callback.match === "string" && String(data || "").startsWith(callback.match)) {
                return { callback, match: null };
            }
        }

        return null;
    }
}

 * Singleton instance of the CommandRegistry.
 * @type {CommandRegistry}
 */
export const registry = new CommandRegistry();
